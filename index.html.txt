  <script>
    (function(){
      const root=document.documentElement;
      const stored=localStorage.getItem('ms-theme');
      if(stored) root.setAttribute('data-theme',stored);
      document.getElementById('themeToggle').addEventListener('click',()=>{
        const next=root.getAttribute('data-theme')==='dark'?'':'dark';
        if(next) root.setAttribute('data-theme','dark'); else root.removeAttribute('data-theme');
        localStorage.setItem('ms-theme',next);
      });
      const posts=Array.from(document.querySelectorAll('.post'));
      const searchInput=document.getElementById('searchInput');
      const clearBtn=document.getElementById('clearSearch');
      const filters=Array.from(document.querySelectorAll('.filter'));
      const langRadios=document.querySelectorAll('input[name="lang"]');
      function applyFilters(){
        const q=(searchInput.value||'').toLowerCase().trim();
        const lang=document.querySelector('input[name="lang"]:checked').value;
        const activeTags=filters.filter(f=>f.checked).map(f=>f.value);
        posts.forEach(p=>{
          const title=p.dataset.title.toLowerCase();
          const tags=(p.dataset.tags||'').toLowerCase().split(',');
          const postLang=p.dataset.lang||'en';
          const matchesQuery=q===''||title.includes(q)||p.innerText.toLowerCase().includes(q);
          const matchesLang=lang==='all'||postLang===lang;
          const matchesTags=activeTags.length===0||tags.some(t=>activeTags.includes(t));
          p.style.display=(matchesQuery&&matchesLang&&matchesTags)?'':'none';
        });
      }
      searchInput.addEventListener('input',applyFilters);
      clearBtn.addEventListener('click',()=>{searchInput.value='';applyFilters();});
      filters.forEach(f=>f.addEventListener('change',applyFilters));
      langRadios.forEach(r=>r.addEventListener('change',applyFilters));
      applyFilters();
    })();
  </script>
